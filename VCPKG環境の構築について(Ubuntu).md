# 前提ディストリビューション
- Ubuntu 25.10 

# OS機能のインストール
```bash
sudo apt update

# ビルドツール一式
sudo apt install -y \
  build-essential \
  cmake \
  git \
  pkg-config

# 必要ライブラリ一式
sudo apt install -y \
  libgl-dev \
  libsdl3-dev \
  libsdl3-image-dev \
  libsdl3-ttf-dev \
  libassimp-dev \
  libopenal-dev \
  libfmt-dev \
  libmpg123-dev
```
